\begin{itemize} \itemsep=5ex
\item $\sin 2\theta= 2 \sin \theta \cos \theta$
\item $\sin^2 \theta= \dfrac{1 - \cos 2\theta}{2}$
\item $\cos^2 \theta= \dfrac{1 + \cos 2\theta}{2}$
\item $\sin^2 \theta + \cos^2 \theta= 1$ \par
	\begin{tikzpicture}

	\node[align=left] (F) at (12,-2) {$\sim$ \enskip $\cot^2 \theta= \csc^2 \theta - 1$ \\[0.2cm] $\sim$ \enskip $\csc^2 \theta= \cot^2 \theta + 1$};
