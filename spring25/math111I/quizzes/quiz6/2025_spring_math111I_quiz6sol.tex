% Problem 1
\problem Showing all your work, factor the following:
	\begin{enumerate}[(a)]
	\item $2x^2 + 4x - 48$
	\item $4x^2 - 25$
	\item $2x^2 - 5x - 3$
	\end{enumerate} 

\ans{
\begin{enumerate}[(a)]
\item First, observe that we can factor out 2: $2x^2 + 4x - 48= 2(x^2 + 2x - 24)$. We need to find factors of $-24$ that add to $2$. 
	\begin{table}[!ht]
	\centering
	\underline{\bfseries $\mathbf{-24}$} \pvspace{0.2cm}
	\begin{tabular}{rr}
	$1 \cdot -24$ & $-23$ \\
	$-1 \cdot 24$ & $23$ \\
	$2 \cdot -12$ & $-10$ \\
	$-2 \cdot 12$ & $10$ \\
	$3 \cdot -8$ & $-5$ \\
	$-3 \cdot 8$ & $5$ \\
	$4 \cdot -6$ & $-2$ \\
	\underline{$\mathbf{-4 \cdot 6}$} & \underline{$\mathbf{2}$} \\
	\end{tabular}
	\end{table}
Therefore, we have\dots
	\[
	2x^2 + 4x - 48= 2(x^2 + 2x - 24)= 2(x - 4)(x + 6)
	\]

\item This is a difference of perfect squares: $a^2 - b^2= (a - b)(a + b)$. We have $a= 2x$, so that $a^2= (2x)^2= 4x^2$, and $b= 5$, so that $b^2= 25$. But then\dots
	\[
	4x^2 - 25= (2x - 5)(2x + 5)
	\]

\item We find that $ac= 2 \cdot -3= -6$. We need to find factors of $-6$ that add to $b= -5$. 
	\begin{table}[!ht]
	\centering
	\underline{\bfseries $\mathbf{-6}$} \pvspace{0.2cm}
	\begin{tabular}{rr}
	\underline{$\mathbf{1 \cdot -6}$} & \underline{$\mathbf{-5}$} \\
	$-1 \cdot 6$ & $5$ \\
	$2 \cdot -3$ & $-1$ \\
	$-2 \cdot 3$ & $1$
	\end{tabular}
	\end{table}
But then\dots
	\[
	2x^2 - 5x - 3= 2x^2 + x - 6x - 3= x(2x + 1) - 3(2x + 1)= (2x + 1)(x - 3)
	\]
\end{enumerate}
}



% Problem 2
\problem Without explicitly attempting to factor $x^2 + 5x + 3$, explain why it cannot factor `nicely.' \pspace

\ans{We know that a quadratic function factors if and only if the discriminant is a perfect square. The discriminant of a quadratic function $ax^2 + bx + c$ is $D= b^2 - 4ac$. We know for $x^2 + 5x + 3$ that $D= 5^2 - 4(1)3= 25 - 12= 13$. But $\sqrt{13} \approx 3.60555$ is not a perfect square. Therefore, $x^2 + 5x + 3$ cannot factor `nicely.'}
